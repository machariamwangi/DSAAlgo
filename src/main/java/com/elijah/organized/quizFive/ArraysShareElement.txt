int[] numsArray1 =  new int[] {6,7,2,9};
int[] numsArray2 = new int[] {1,2,3,4};

public bool arraysShareElement(numsArray1, numsArray2)
{
   for(int i =0; i < numsArray1.length; i++) {  //O(1)
      for(int j =0 ; j< numsArray2.length; j++) {  // O(1)
         if(numsArray1[1] == numsArray2[j]){
           return true
         }
      }
   }

   //if nothing matches
   return false;
}
