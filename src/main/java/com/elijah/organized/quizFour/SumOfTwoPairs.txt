public object SumOfTwoPairst(int[] numbers, int target)
{
 //we have two holder i and J which as constant (0(1)

 //the for loop to return will have to run n times * n times ie. if array length is 10 they will run 100 times

  // n^2 * O(1)
 //
  for(int i = 0; i < numbers.length; i++)
  {
    for(int j = i + 1; j< numbers.length; j++)
    {
       if (numbers[i] + numbers[j] == target)
        {
          // Return the pair as an int array
           return new int[]{numbers[i], numbers[j]};  // this will make the loop to terminate  incase the answer is found  // O(1)
         }
    }
  }

  // if no solution
   return "no solution"; // O(1)

   //if we remove coeffiecient we will have O(n^2)
   // space conflexity does change so we have O(1)
}
